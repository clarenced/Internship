\def\maketitle{%
  \thispagestyle{empty}\vbox to \vsize{%
    \vfill
    \vspace{1cm}
    \begin{flushleft}
      \usefont{OT1}{ptm}{m}{n}
      \huge \@title

    \par
    \hrule height 4pt
    \par

      \par
    \end{flushright}
    \vspace{1cm}
    \vfill
    \vfill
    \bf{Le \@date}
  }%
  \cleardoublepage
}
